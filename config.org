(defun fanshi/clone-from-project (dir)
  "Clone the infrastructure of an existing project DIR to make a new project"
  (require 'projectile)
  (if-let ((project-type (projectile-project-type dir))
           (project-files-to-copy (symbol-value (intern-soft (concat "fanshi/project-files-to-copy/" (prin1-to-string project-type)))))
           (new-project-directory (file-name-as-directory (read-directory-name "Create new project at directory: "))))
      (progn
        ;; NOTE: make new directory
        (make-directory new-project-directory t)
        ;; NOTE: copy files
        (dolist (wildcards-or-file project-files-to-copy)
          (dolist (file (projectile-verify-file-wildcard wildcards-or-file dir))
                  (dired-copy-file file new-project-directory 1)))
        ;; NOTE: init project
        (fanshi/init-new-project new-project-directory)
        ;; NOTE: switch to the new project
        (projectile-switch-project-by-name new-project-directory))
    (user-error (concat "Unsupported project type to clones: " (prin1-to-string project) " + " (prin1-to-string project-type) " + " (prin1-to-string project-files-to-copy)))))

** Fix response org heading level
Disable due to it seems to mess thing up when I use `fanshi/multisend'
- ref :: https://github.com/karthink/gptel/issues/1226#issuecomment-3828162897
#+begin_src emacs-lisp :tangle no
(after! gptel
  (defun gptel-fix-response-levels (beg end)
    "Correctly shifts heading levels ONLY within the gptel response."
    (save-excursion
      (let ((parent-level
             ;; 1. Find the parent level by looking BEFORE the response (beg)
             (save-excursion
               (goto-char beg)
               (if (re-search-backward org-heading-regexp nil t)
                   (org-current-level)
                 0)))) ;; Default to 0 if at top of file

        (save-restriction
          ;; 2. CRITICAL: Lock operations to ONLY the response text
          (narrow-to-region beg end)
          (goto-char (point-min))

          ;; 3. Check if the response actually has headings
          (when (re-search-forward org-heading-regexp nil t)
            (let* ((current-level (org-current-level))
                   ;; We want the response to be 1 level deeper than parent
                   (target-level (1+ parent-level))
                   (diff (- target-level current-level)))

              (unless (zerop diff)
                ;; Apply shift to ALL headings in the narrowed response
                (goto-char (point-min))
                (while (re-search-forward "^\\(\\*+\\) " nil t)
                  (let* ((stars (match-string 1))
                         (new-level (max 1 (+ (length stars) diff))))
                    (replace-match (concat (make-string new-level ?*) " ") t t)))
                (message "Gptel: Adjusted response depth by %d." diff))))))))

  (add-hook 'gptel-post-response-functions #'gptel-fix-response-levels))

                        :callback (lambda (resp _info)
                                    (setq classification resp)
                                    (setq done t))
                        :fsm inner-fsm)
                      (while (and (not done) (> retry-quota 0))
                        (sleep-for 0 200)))
                  (error
                   (message "gptel: truncation check failed (retry %d): %s"
                            retry-quota err)
                   (setq classification nil))))
              (if (or (not (stringp classification))
                      (string= (downcase (string-trim classification)) "complete"))
                  (fanshi/notify)
                (message "gptel: Response appears truncated, continuing...")
                (let ((fanshi/prevent-duplicate-gptel-send nil)) (gptel-send))))))
      (fanshi/notify)))

(after! gptel
  (defun fanshi/gptel-agent--helpful-symbol (symbol-name &optional force-variable)
    "Return a curated help buffer content for SYMBOL-NAME as a string.

Unlike helpful-update which builds UI-oriented sections (Key Bindings,
Debugging buttons, References scanning, Symbol Properties), this builds
only the sections useful for an AI agent:

  - Summary: one-line description
  - Obsolete Info: if the symbol is deprecated
  - Signature: for functions/macros/commands
  - Value / Global Value: for variables
  - Original Value: if the variable value has been modified
  - Documentation: full docstring with version info
  - Advice: whether the function has advice attached
  - Aliases: other names pointing to the same symbol
  - Implementations: for generic methods
  - Source Code: the actual elisp source

Symbol-Name is a string naming an Elisp function, variable, macro, or face."
    (if-let ((symbol (intern-soft symbol-name)))
        (progn
          (when (and force-variable (not (boundp symbol)) (fboundp symbol))
            (error "Error: force-variable is t but %s is bounded only with function" symbol))
          ;; Determine callable-p: prefer force-variable if set, else heuristic.
          (-let* ((callable-p (if force-variable
                                  nil
                                (fboundp symbol)))
                  ;; Pre-compute values needed for building sections.
                  (primitive-p (helpful--primitive-p symbol callable-p))
                  ((def-buf def-pos _opened)
                   (if (or (not primitive-p) find-function-C-source-directory)
                       (helpful--definition symbol callable-p)
                     '(nil nil nil)))
                  (source (when def-buf
                            (helpful--source symbol callable-p def-buf def-pos)))
                  (source-path (when def-buf (buffer-file-name def-buf))))
            ;; Build output in a temp buffer to avoid polluting the helpful buffer.
            (with-temp-buffer
              ;; --- Summary ---
              (insert (helpful--summary symbol callable-p def-buf def-pos) "\n\n")

              ;; --- Obsolete Info ---
              (when (helpful--obsolete-info symbol callable-p)
                (insert (helpful--format-obsolete-info symbol callable-p) "\n\n"))

              ;; --- Signature (callables only, skip keyboard macros) ---
              (when (and callable-p (not (helpful--kbd-macro-p symbol)))
                (insert "Signature\n")
                (insert (helpful--syntax-highlight (helpful--signature symbol)) "\n\n"))

              ;; --- Value / Global Value (variables only) ---
              (when (not callable-p)
                (let* ((val (helpful--sym-value symbol nil))
                       (heading-text
                        (if (local-variable-if-set-p symbol)
                            "Global Value"
                          "Value")))
                  (insert heading-text "\n")
                  (insert (helpful--format-value symbol val) "\n\n")
                  ;; Original Value if it differs from current.
                  (when (helpful--original-value-differs-p symbol)
                    (insert "Original Value\n")
                    (insert (helpful--format-value symbol (car (helpful--original-value symbol))) "\n\n"))))

              ;; --- Documentation ---
              (let ((docstring (helpful--docstring symbol callable-p))
                    (version-info (unless callable-p
                                    (helpful--version-info symbol))))
                (when (or docstring version-info)
                  (insert "Documentation\n")
                  (when docstring
                    (insert (helpful--format-docstring docstring) "\n"))
                  (when version-info
                    (insert (s-word-wrap 70 version-info) "\n"))))

              ;; --- Advice ---
              (when (helpful--advised-p symbol)
                (insert "Advice\n")
                (insert (format "This %s is advised."
                                (if (macrop symbol) "macro" "function")) "\n\n"))

              ;; --- Aliases ---
              (when-let ((aliases (helpful--aliases symbol callable-p)))
                (insert "Aliases\n")
                (insert (s-join "\n" (--map (helpful--format-alias it callable-p) aliases))
                        "\n\n"))

              ;; --- Implementations (generic methods) ---
              (when callable-p
                (ignore-errors
                  (let ((func symbol)
                        (content))
                    (when (fboundp #'cl--generic-describe)
                      (with-temp-buffer
                        (declare-function cl--generic-describe "cl-generic" (function))
                        (cl--generic-describe func)
                        (goto-char (point-min))
                        (when (re-search-forward "^Implementations:$" nil t)
                          (setq content (buffer-substring (point) (point-max))))))
                    (when content
                      (insert "Implementations\n")
                      (insert (s-trim content) "\n\n")))))

              ;; --- Source Code ---
              (when (or source-path primitive-p)
                (insert "Source Code\n")
                (when source-path
                  (insert (format ";; Defined in %s\n" source-path)))
                (when (and primitive-p (not source-path))
                  (insert "C code is not yet loaded.\n")))
              (when source
                (cond
                 ((stringp source)
                  (insert (helpful--syntax-highlight source)))
                 ((and (consp source) (eq (car source) 'closure))
                  (insert (helpful--syntax-highlight
                           (concat ";; Closure converted to defun by helpful.\n"
                                   (helpful--pretty-print
                                    (helpful--format-closure symbol source))))))
                 (t
                  (insert (helpful--syntax-highlight
                           (concat ";; Could not find source code, showing raw function object.\n"
                                   (helpful--pretty-print source))))))
                (insert "\n"))
              (buffer-substring-no-properties (point-min) (point-max)))))
      (error "Error: Unknown symbol %s" symbol-name)))

(defun fanshi/ai-imenu--flatten (items &optional parent)
  "Flatten imenu tree ITEMS into a list of (name . position) pairs.

ITEMS is a list of items, where each item is either a pair (NAME . POSITION)
or a pair (NAME . CHILDREN) with CHILDREN being a list of items.

PARENT is the name of the parent section, used to build hierarchical
names by joining parent and child names with a separator."
  (mapcan
   (lambda (item)
     (let* ((name (car item))
            (rest (cdr item))
            (parent (if parent (concat parent " / " name) name)))
       (if (listp rest)
           (fanshi/ai-imenu--flatten (cdr item) parent)
         ;; (or (markerp rest) (integerp rest))
         (list (cons parent rest)))))
   items))

                          (goto-char pos)

                                  name
                                  (line-number-at-pos nil t)

#+begin_src emacs-lisp :tangle yes
(after! gptel
  (defun fanshi/gptel--validate-simple-tool-arg-type (type arg)
    (pcase type
      ((or 'null "null") (null arg))
      ((or 'string "string") (stringp arg))
      ((or 'number "number") (numberp arg))
      ((or 'integer "integer") (integerp arg))
      ((or 'boolean "boolean") (or (and (booleanp arg) (eq t arg)) (eq :json-false arg)))
      (_ (progn (message "Warning: FIXME tool call validation skipped for type %s: %s" type arg) t))))

  (defun fanshi/gptel--validate-tool-args (tool-spec args)
    "Validate ARGS according to TOOL-SPEC.

TOOL-SPEC is a `gptel-tool' and ARGS is a plist of arguments for a tool
call.  The argument list is suitable for supplying to the tool function.
Returns nil if validation succeeded, else a string describing errors."

    (let* ((tool-spec-args (gptel-tool-args tool-spec))
           (expected-args (mapcar (lambda (pl) (plist-get pl :name)) tool-spec-args))
           (unepected-arg-errs
            (mapconcat
             (lambda (input-arg)
               (let ((tool-name (s-chop-prefix ":" (symbol-name input-arg))) )
                 (unless (member tool-name expected-args)
                   (format "Error: tool call arg %s is not expected.\n" tool-name))))
             (doom-plist-keys args) "")))
      (if (string-empty-p unepected-arg-errs)
          (mapconcat
           (lambda (spec-arg)
             (let* ((name (plist-get spec-arg :name))
                    (input_arg (plist-get args (intern (concat ":" name))))
                    (description (plist-get spec-arg :description))
                    (optional (plist-get spec-arg :optional))
                    (enum (plist-get spec-arg :enum))
                    (type (plist-get spec-arg :type))
                    (simple-error (format "Error: tool call arg %s (%s) expected type %s but got %s\n\n" name description type input_arg)))
               (unless (and optional (null input_arg))
                 (cond
                  ((not (fanshi/gptel--validate-simple-tool-arg-type type input_arg)) simple-error)
                  ((and enum (null (seq-position enum input_arg))) (format "Error: tool call arg %s (%s) expected enum (in %s) but got %s\n\n" name description enum input_arg))
                  ((and (eq type 'array) (not (listp input_arg))) simple-error)
                  ((eq type 'array) (let ((items (plist-get spec-arg :items)))
                                      (when (and items (all (lambda (item) (not (fanshi/gptel--validate-simple-tool-arg-type item input_arg optional))) items))
                                        (format "Error: tool call arg %s (%s) expected array with item type in %s but got %s\n\n" name description items input_arg))))
                  ;; TEMP FIXME: I don't have a tool with object arg yet, so skip it now
                  ((eq type 'object) nil)
                  (t nil)))))
           (gptel-tool-args tool-spec) "")
        (format "%s\n expected %s" unepected-arg-errs expected-args))))

  (defun fanshi/gptel--validate-tool-args-hook (tool-call)
    "Validate TOOL-CALL args via `gptel-pre-tool-call-functions'.

Returns a plist with :BLOCK on validation failure, or nil to proceed."
    (if-let ((name (plist-get tool-call :name)))
        (if-let ((tool-spec (gptel-get-tool name)))
            (let* ((errors (fanshi/gptel--validate-tool-args tool-spec (plist-get tool-call :args))))
              (if (not (string= errors ""))
                  `(:block ,errors)
                (when (member name fanshi/safe-tool) `(:confirm nil))))
          `(:block ,(format "Error: Unknown tool called: %s" name)))
      `(:block "Error: tool called with empty name")))

  (add-to-list 'gptel-pre-tool-call-functions #'fanshi/gptel--validate-tool-args-hook))

(after! gptel
  (defun fanshi/gptel--handle-post-tool-a (fsm)
    "Run `gptel-post-tool-call-functions for FSM."
    (let* ((info (gptel-fsm-info fsm))
           (buffer (plist-get info :buffer)))
      (when (buffer-local-value 'gptel-post-tool-call-functions buffer)
        (let ((hook-func-args (list :buffer (buffer-name buffer)
                                    :backend (plist-get info :backend)
                                    :model (plist-get info :model))))
          (with-current-buffer buffer
            (run-hook-wrapped           ; Run pre tool call functions
             'gptel-post-tool-call-functions
             (lambda (hook-func)
               (prog1 nil
                 (dolist (tool-call (plist-get info :tool-use))
                   (let* ((name (plist-get tool-call :name))
                          (args (plist-get tool-call :args))
                          (hook-func-result
                           (with-demoted-errors "gptel-post-tool-call hook error: %S"
                             (funcall hook-func
                                      (nconc (list :name name :args args
                                                   :result (plist-get tool-call :result))
                                             hook-func-args)))))
                     (if (plist-get hook-func-result :stop)
                         (let ((reason (or (plist-get hook-func-result :stop-reason)
                                           (concat "Request stopped by post-tool-call hook "
                                                   (and (symbolp hook-func)
                                                        (symbol-name hook-func))
                                                   " (tool \"" name "\")"))))
                           (plist-put info :stop-reason reason)
                           (plist-put info :status "Stopped by hook")
                           (plist-put info :error reason))
                       ;; TODO(tool-hooks): :block behavior not final!
                       (let ((blockp (plist-get hook-func-result :block))
                             (result (plist-get hook-func-result :result)))
                         (when blockp
                           (plist-put tool-call :error t)
                           (setq result
                                 (concat "Error: "
                                         (if (stringp blockp) blockp
                                           (format "Tool %s blocked by user" name)))))
                         (when result
                           (cl-loop     ; Update results sent to callback
                            for call in (plist-get info :tool-result)
                            for (spec stored-args _) = call
                            when (null spec)
                            ;; TEMP HACK when the agent called with a nonexistent tool, we make a fake tool here so that the ui doesn't break
                            do (setf (car call) (gptel-make-tool :name (intern name)))
                            when (and
                                  (or (null spec)
                                      (equal (gptel-tool-name spec) name))
                                  (null (cl-set-difference
                                         stored-args args :test #'equal)))
                            do (setf (caddr call) result) and return nil
                            finally
                            (display-warning
                             '(gptel tools)
                             (format "Tool %s: Could not replace tool results" name)))
                           ;; Update results sent to LLM
                           (plist-put tool-call :result result))))))))))))))

*** Tool Call Failures removal
- why :: remove those noises from the context. When gptel-agent retries a tool call
  (wrong args or non-existing tool), the failed attempts pollute the context with
  error messages. We clean them up automatically after a successful retry.
- approach :: Use =gptel-prompt-transform-functions= to mark failed blocks with
  the text property ='ignore=. gptel's =gptel--parse-buffer= naturally skips
  regions with ='ignore, so the LLM never sees them. The blocks remain visible
  in the buffer for debugging.
- removal criterion :: Failed blocks that precede a successful call of the same tool name
  (consecutive failures between them, ignoring other tool calls in between)
- algorithm :: On each *successful* tool call, look backward to find the streak of
  same-tool failures immediately preceding it. Those failures are hidden.
- additional cases considered:
  - Tool blocked by user confirmation timeout :: Already shows "Error: ", covered.
  - Failed calls to DIFFERENT tools :: NOT removed (conservative — different tools may
    carry useful debugging context for the LLM)
  - Failed calls NOT followed by success :: NOT removed (LLM needs to see the error)
#+begin_src emacs-lisp :tangle yes
(after! gptel
  (defcustom fanshi/gptel-clean-failed-tool-calls t
    "Whether to automatically hide failed tool call blocks from LLM context.

When non-nil, consecutive failed tool calls (wrong args or unknown tools)
that precede a successful call of the same tool are hidden from the LLM
by setting their text property to 'ignore.

The blocks remain visible in the buffer for debugging, but are skipped
when gptel parses the prompt to send to the LLM."
    :type 'boolean
    :group 'gptel)

  ;; Prompt transform: on each successful tool call, search backward for
  ;; same-tool failures and mark them with 'ignore so gptel--parse-buffer skips them.
  (defun fanshi/gptel--tool-call-failure-hide-transform (fsm)
    "Hide failed tool call blocks by marking them with gptel 'ignore property.

gptel--parse-buffer skips regions with gptel 'ignore, so these blocks
are invisible to the LLM while remaining visible in the buffer.

Additionally, marks the corresponding blocks in the original gptel buffer
so that the 'ignore persists across requests — no redundant re-scanning."
    (when fanshi/gptel-clean-failed-tool-calls
      (save-excursion
        (goto-char (point-max))
        (let ((successful-tool-call-name))
          (while (/= (point) (point-min))
            (let ((end (point))
                  (start (goto-char (previous-single-property-change (point) 'gptel nil (point-min)))))
              (pcase (get-char-property (point) 'gptel)
                (`(tool . ,id)
                 (condition-case nil
                     (save-excursion
                       (let* ((tool-call (read (current-buffer)))
                              (name (plist-get tool-call :name)))
                         (if (re-search-forward (rx (and point
                                                         (* (syntax -))
                                                         (optional (and "error" (+ (syntax -))))
                                                         (or "<tool_call_error>" "Error:")))
                                                nil t)
                             ;; Failed call: hide it if we already saw a
                             ;; successful call of the same tool name.
                             (when (and successful-tool-call-name
                                        (string= successful-tool-call-name name))
                               ;; Mark in temp buffer (for this request).
                               (put-text-property start end 'gptel 'ignore)
                               ;; ALSO mark in original buffer (persists across requests).
                               (when-let* ((original-buffer (plist-get (gptel-fsm-info fsm) :buffer))
                                           (orig-buf (get-buffer original-buffer)))
                                 (with-current-buffer orig-buf
                                   (save-excursion
                                     (goto-char (point-max))
                                     (while (not (or (equal (point) (point-min))
                                                     (equal (get-char-property (point) 'gptel)
                                                            `(tool . ,id))))
                                       (goto-char (previous-single-property-change (point) 'gptel nil (point-min))))
                                     (if (equal (get-char-property (point) 'gptel)
                                                `(tool . ,id))
                                         (let ((prop-end (save-excursion (next-single-property-change (point) 'gptel nil (point-max)))))
                                           (put-text-property (point) prop-end 'gptel 'ignore))
                                       (warn "tool-call(%s) not found in %s" id original-buffer))))))
                           ;; Successful call: record the tool name.
                           (setq successful-tool-call-name name))))
                   (t
                    (message (format "fanshi/gptel--tool-call-failure-hide-transform: could not parse tool-call %s on line %s"
                                     id (line-number-at-pos (point)))))))
                (_ nil)))
            (ignore-errors (backward-char)))))))

  ;; Register the prompt transform.
  (add-to-list 'gptel-prompt-transform-functions
               #'fanshi/gptel--tool-call-failure-hide-transform))

(defun fanshi/gptel-make-openai (backend-name host &optional pass-entry protocol set-as-backend)
  "Register an OpenAI-compatible gptel backend.

Fetches available models from HOST via /v1/models, then calls
`gptel-make-openai' with the discovered model list.

BACKEND-NAME is the name of the new backend.

HOST is the server hostname (with optional port), e.g.
\"127.0.0.1:8080\" or \"integrate.api.nvidia.com\".

PASS-ENTRY is an optional auth secret entry passed to
`+pass-get-secret'.  If it is a string, it is treated as the
entry name; if it is nil or non-string, it is used directly as
the :key for `gptel-make-openai'.

PROTOCOL is the URL scheme, defaulting to \"http\".  Use \"https\"
for remote APIs."
  (let* ((url-request-method "GET")
         (url-request-data nil)
         (pass-key (when (and pass-entry (stringp pass-entry)) (+pass-get-secret pass-entry)))
         (url-request-extra-headers (if pass-key
                                        `(("Authorization" . ,(concat "Bearer " pass-key))
                                          ,@url-request-extra-headers)
                                      url-request-extra-headers))
         (protocol (or protocol "http")))
    (url-retrieve (concat protocol "://" host "/v1/models")
                  (lambda (status)
                    (if (eq (car status) :error)
                        (user-error (format "%s" (cdr status)))
                      (goto-char url-http-end-of-headers)
                      (let* ((json-object-type 'alist)
                             (json-array-type 'list)
                             (backend (gptel-make-openai backend-name
                                        :host host
                                        :protocol protocol
                                        :key (or pass-key (unless (stringp pass-entry) pass-entry))
                                        :stream t
                                        :models (cl-loop
                                                 for model in (alist-get 'data (json-read))
                                                 for id = (alist-get 'id model)
                                                 ;; unless (string-match "dflash" id)
                                                 ;; unless (string-match "Kokoro" id)
                                                 ;; unless (string-match "VoxCPM2" id)
                                                 ;; unless (string-match "fish-audio" id)
                                                 ;; unless (string-match "Wan2.2" id)
                                                 collect (intern id)))))
                        (when set-as-backend (setopt gptel-backend backend))))))))

  (defun fanshi/gptel--load-models-from-ini (backend)
    "Parse INI file for BACKEND ('Llama or 'Mlx) and return model specifications for gptel.
For 'Llama: uses section name as model symbol, :hf tag for capability inference.
For 'Mlx: uses :hf tag value as model symbol."
    (with-temp-buffer
      (insert-file-contents-literally (expand-file-name (concat (symbol-name backend) "-model.ini") doom-user-dir))
      (goto-char (point-min))
      (let ((models '()))
        (while-let ((_ (not (eobp)))
                    (line (buffer-substring-no-properties (line-beginning-position) (line-end-position))))
          (if ;; Section header: [model-name]
              (string-match "^\\[\\([^]]+\\)\\]$" line)
              (let ((section (intern (match-string 1 line))) (props))
                (forward-line 1)
                (while-let ((_ (not (eobp)))
                            (line (buffer-substring-no-properties (line-beginning-position) (line-end-position)))
                            (_ (not (string-match "^\\[\\([^]]+\\)\\]$" line))))
                  (when ;; Property line: key = value
                      (string-match "^\\([a-z-]+\\)\\s-*=\\s-*\\(.*\\)$" line)

                                 ("temp" :temperature)
                                 ("ctx-size" :context-window)
                                 (str (intern (concat ":" (string-replace "-" "_" str))))))
                          (val (pcase (match-string 2 line)
                                 ((and (pred (string-match-p "^[0-9.]+$")) str) (string-to-number str))
                                 (str str))))
                      (if (memq key '(:temperature :top_p :top_k :min_p :presence_penalty))
                          (let ((request-params (plist-put (plist-get props :request-params) key val)))
                            (setf (plist-get props :request-params) request-params))
                        (setq props (plist-put props key val)))))
                  (forward-line 1))
                (let (;; For Mlx, use :hf tag value as the model symbol
                      (model-symbol  (or (and (eq backend 'Mlx) (intern (plist-get props :hf)))
                                         section))
                      (caps-mime (fanshi/gptel--hf-to-capabilites-and-mimes (plist-get props :hf))))

                                               (plist-put :output-cost 0)
                                               (plist-put :capabilities (car caps-mime))
                                               (plist-put :mime-types (cdr caps-mime))))
                        models)))
            (forward-line 1)))

(after! gptel
  (defun fanshi/gptel-agent--suggest-filename-prefix ()
    "Suggest a concise filename prefix via gptel-request based on CHAT content.

Runs asynchronously.  When called interactively:
- In a buffer visiting a file: operates on that file.
- In a Dired buffer: operates on marked files if any, otherwise the file at point.

When run from =after-save-hook', CONTENT and FILE are provided as arguments.

If the file already has a prefix (matching pattern PREFIX-TIMESTAMP.org),
replaces the old prefix with the new one."
    (interactive)
    (let ((resolved
           (cond
            ;; Dired buffer — marked files, or file at point
            ((derived-mode-p 'dired-mode)
             (or
              (cl-loop for file in (dired-get-marked-files nil)
                       collect (cons file
                                     (with-temp-buffer
                                       (insert-file-contents file)
                                       (fanshi/gptel-agent--extract-chat-content))))
              (let ((file (dired-get-filename nil t)))
                (if (null file) (user-error "No file at point and no marked files in Dired")
                  (list (cons file (with-temp-buffer (insert-file-contents file) (fanshi/gptel-agent--extract-chat-content))))))))
            ;; Buffer visiting a file
            (buffer-file-name (list (cons buffer-file-name (fanshi/gptel-agent--extract-chat-content))))
            ;; No valid context
            (t (user-error "No file context: not visiting a file and not in Dired")))))
      (cl-loop for res in resolved
               do (fanshi/gptel-agent--suggest-filename-prefix--1 res))))

  (defun fanshi/gptel-agent--suggest-filename-prefix--1 (pair)
    "Process a single (FILE . CONTENT) resolved pair."
    (when-let* ((file (car pair))
                (dir (file-name-directory file))
                (base (file-name-sans-extension file))
                (ext (file-name-extension file))
                ;; Extract trailing timestamp YYYYMMDD-HHMMSS
                (timestamp (when-let*
                               ((match (string-match
                                        "\\([0-9]\\{8\\}-[0-9]\\{6\\}\\)$" base)))
                             (match-string 1 base)))
                (suffix (or timestamp base))
                (content (cdr pair))
                (gptel-model (or (fanshi/gptel--get-active-main-model) 'mlx-community--Qwen3.6-35B-A3B-bf16)))
      (gptel-request (format "%s" content)
        :context nil
        :system "You are a file naming assistant. Based on the following chat transcript, suggest a CONCISE filename prefix (2-4 words). Rules:
- Return ONLY the prefix, nothing else
- Use lowercase, replace spaces with hyphens
- Capture the core topic or task
- No quotes, no code blocks, no explanations"
        :callback (lambda (response _)
                    (when (stringp response)
                      (if-let* ((prefix (downcase (string-trim response)))
                                (prefix (replace-regexp-in-string "[^a-z0-9-]" "-" prefix))
                                (prefix (replace-regexp-in-string "-\\+" "-" prefix))
                                (prefix (string-trim prefix "-"))
                                (new-base (when (> (length prefix) 1) (concat prefix "-" suffix)))
                                (new-file (expand-file-name (concat new-base "." ext) dir)))
                          (rename-file file new-file)
                          (when-let* ((buf (get-file-buffer file)))
                            (with-current-buffer buf
                              (setq buffer-file-name new-file)
                              (rename-buffer (file-name-nondirectory new-file) t)))
                        (message "Failed to rename %s (file may have been moved or deleted)" file))))))))

      (funcall abort-fn)

      (gptel--fsm-transition fsm 'ABRT)
      (message "Stopped gptel request at point in buffer %S"
               (buffer-name buf))
      t))

** Agent chain
Orchestrate multiple gptel presets sequentially, with fan-out for parallel exploration.
Each step uses a preset + prompt template with {{key}} substitution from prior results.
Uses gptel-request directly (no tool access — suitable for analysis/planning chains).
*** fanshi/gptel-chain
#+begin_src emacs-lisp :tangle no :lexical t
(after! gptel
  (defun fanshi/gptel-chain--substitute (template context)
    "Replace {{KEY}} in TEMPLATE with values from CONTEXT (an alist)."
    (let ((result template))
      (dolist (pair context)
        (let* ((key (car pair))
               (placeholder (format "{{%s}}" key)))
          (setq result (replace-regexp-in-string
                        (regexp-quote placeholder)
                        (or (cdr pair) "")
                        result t t))))
      result))

  (cl-defun fanshi/gptel-chain (steps &key context callback on-step)
    "Execute STEPS as a sequential chain of gptel requests.

STEPS is a list of step plists.  Each step has:
  :name       — string key for storing the result
  :preset     — preset symbol (e.g. 'research 'product)
  :prompt     — prompt template with {{key}} placeholders
  :fan-out    — (optional) integer N for parallel requests
  :consolidate — (optional) prompt to merge fan-out results;
                 uses {{responses}} for all results joined by ---
  :model      — (optional) override model
  :backend    — (optional) override backend
  :system     — (optional) override system prompt
  :request-params — (optional) override request params

CONTEXT is an initial alist of (KEY . VALUE) for template substitution.
CALLBACK is called with (FINAL-CONTEXT) when the chain completes.
ON-STEP is called with (STEP-INDEX STEP-NAME RESULT) after each step."
    (let ((ctx (or context '()))
          (remaining steps)
          (idx 0)
          (cb callback)
          (on-step-fn on-step))

      (cl-labels ((log (msg) (message "gptel-chain: %s" msg))

                  (resolve-preset-info (step)
                    "Extract model/backend/system/params from STEP's preset."
                    (let* ((pname (plist-get step :preset))
                           (preset (gptel-get-preset pname)))
                      (unless preset
                        (error "gptel-chain: Unknown preset '%s'" pname))
                      (let ((pp (gptel--preset-plist preset)))
                        (list :model (or (plist-get step :model)
                                         (and pp (plist-get pp :model))
                                         gptel-model)
                              :backend (or (plist-get step :backend)
                                           (and pp (plist-get pp :backend))
                                           gptel-backend)
                              :system (or (plist-get step :system)
                                          (and pp
                                               (let ((s (plist-get pp :system)))
                                                 (if (functionp s) (funcall s) s))))
                              :params (or (plist-get step :request-params)
                                          (and pp (plist-get pp :request-params)))))))

                  (advance (result)
                    "Store RESULT and move to next step."
                    (let ((sname (plist-get (car remaining) :name)))
                      (push (cons sname result) ctx)
                      (log (format "Step '%s' complete" sname))
                      (when on-step-fn
                        (funcall on-step-fn idx sname result)))
                    (setq remaining (cdr remaining)
                          idx (1+ idx))
                    (if remaining
                        (exec-step (car remaining))
                      (when cb (funcall cb ctx))))

                  (fan-out-done (responses)
                    "Handle fan-out completion: consolidate or concatenate."
                    (let* ((step (car remaining))
                           (consolidate (plist-get step :consolidate))
                           (info (resolve-preset-info step)))
                      (if consolidate
                          (let ((prompt (fanshi/gptel-chain--substitute
                                         consolidate
                                         `(("responses" . ,(string-join responses "\n\n---\n\n"))))))
                            (log "Consolidating fan-out results...")
                            (gptel-request prompt
                              :model (plist-get info :model)
                              :backend (plist-get info :backend)
                              :system (plist-get info :system)
                              :request-params (plist-get info :params)
                              :callback (lambda (resp)
                                          (if (stringp resp)
                                              (advance resp)
                                            (log "Consolidation returned non-string")
                                            (advance "")))))
                        (advance (string-join responses "\n\n---\n\n")))))

                  (exec-step (step)
                    "Execute a single STEP."
                    (let* ((name (plist-get step :name))
                           (info (resolve-preset-info step))
                           (prompt (fanshi/gptel-chain--substitute
                                    (plist-get step :prompt) ctx))
                           (fan-out (plist-get step :fan-out)))
                      (log (format "Step %d: %s" idx name))
                      (if (and fan-out (> fan-out 1))
                          ;; Fan-out: send N parallel requests
                          (let ((responses '())
                                (done 0)
                                (total fan-out))
                            (dotimes (_ fan-out)
                              (gptel-request prompt
                                :model (plist-get info :model)
                                :backend (plist-get info :backend)
                                :system (plist-get info :system)
                                :request-params (plist-get info :params)
                                :callback (lambda (resp)
                                            (when (stringp resp)
                                              (push resp responses))
                                            (cl-incf done)
                                            (when (>= done total)
                                              (fan-out-done (nreverse responses))))))
                            ;; Timeout fallback: proceed with partial results after 3 min
                            (run-at-time 180 nil
                                         (lambda ()
                                           (when (< done total)
                                             (log (format "Fan-out timeout, %d/%d responses" done total))
                                             (fan-out-done (nreverse responses))))))
                        ;; Single request
                        (gptel-request prompt
                          :model (plist-get info :model)
                          :backend (plist-get info :backend)
                          :system (plist-get info :system)
                          :request-params (plist-get info :params)
                          :callback (lambda (resp)
                                      (if (stringp resp)
                                          (advance resp)
                                        (log (format "Step '%s' returned non-string: %s" name resp))
                                        (advance "")))))))

                  (when remaining
                    (exec-step (car remaining)))))))

  ;; Convenience: run a pre-defined chain variable
  (defun fanshi/gptel-chain-run (chain-var &rest context)
    "Run a pre-defined CHAIN-VAR with CONTEXT.
CHAIN-VAR is a variable holding a chain definition.
CONTEXT is an alist of (KEY . VALUE) for template substitution."
    (interactive)

  (defun fanshi/ai-delegate--tool-delegate (task agent task-id)
    "Delegate the task to an agent or to the user.
If AGENT is \"user\", delegate to human for input, using TASK
as the description of what's needed.
Otherwise, spawn a dedicated gptel-agent buffer with AGENT
working on TASK in an isolated jj workspace.
TASK-ID is the org-id of the task entry to track."
    (if (or (string= agent "user") (string= agent 'user))
        ;; Delegate to human
        (if task-id
            (progn
              (org-id-goto task-id)
              (org-entry-put (point) "DELEGATED_TO" "user")
              (org-end-of-subtree t t)
              (insert (format "\n- %s [USER]: %s"
                              (format-time-string "%Y-%m-%d %H:%M")
                              task))
              (save-buffer)
              (message "Task delegated to user: %s" task)
              (format "Task delegated to user: %s" task))
          (error "Cannot delegate to user without task-id"))
      ;; Delegate to AI agent
      (unless task-id (error "task-id is required"))
      (let* ((project-root (projectile-project-root))
             (branch-name (format "task-%s-%s" task-id (format-time-string "%Y%m%d-%H%M%S")))
             (target-dir (expand-file-name (concat ".trees/" branch-name) project-root))
             (child-buf-name (format "*gptel-agent: %s*" (substring branch-name 0 30))))
        ;; 1. Create jj workspace if needed
        (unless (file-directory-p target-dir)
          (make-directory (file-name-directory target-dir) t)
          (with-temp-buffer
            ;; Create a jj workspace for this agent
            (unless (zerop (call-process "jj" nil t nil "workspace" "add" target-dir))
              (error "Failed to create jj workspace at '%s'" target-dir))
            (let ((default-directory target-dir))
              ;; Configure agent identity in the workspace
              (unless (zerop (call-process "jj" nil t nil "config" "set" "--repo" "user.name" (format "Agent (%s)" agent)))
                (error "Failed to create jj config set username '%s' for repo '%s'" agent target-dir))
              (unless (zerop (call-process "jj" nil t nil "config" "set" "--repo" "user.email" (format "%s-agent@local" agent)))
                (error "Failed to create jj config set username '%s' for repo '%s'" agent target-dir))
              ;; Start a new commit on top of main with the task description
              (let (exit-code (call-process "jj" nil t nil "new" "main@" "-m" task))
                (unless (zerop exit-code)
                  (message "Warning: jj new failed with code %d" exit-code))))))
        ;; 2. Update org entry with DELEGATED status and properties
        (save-excursion
          (org-id-goto task-id)
          (org-entry-put (point) "WORKTREE_DIR" target-dir)
          (org-entry-put (point) "DELEGATED_TO" (if (symbolp agent) (symbol-name agent) agent))
          (org-entry-put (point) "DELEGATED_AT" (format-time-string "%Y-%m-%d %H:%M:%S"))
          (org-entry-put (point) "AGENT_BUF" child-buf-name)
          (org-todo "DELEGATED")
          (save-buffer))
        ;; 3. Create and configure the child agent buffer using gptel-agent
        (with-current-buffer (get-buffer-create child-buf-name)
          (gptel-agent target-dir agent)
          (setq-local fanshi/gptel-agent-auto-workspace nil
                      fanshi/ai-delegate--current-task-id task-id)
          (goto-char (point-max))
          (insert (format "# Delegated Task\n\n%s\n\nYou are working in an isolated jj workspace.\nAll file changes are automatically committed to your working-copy commit (@).\nWhen you complete the task, call the FinishTask tool with a summary.\nIf you cannot complete it, call AbortTask with a reason.\nIf you need human input, call Delegate with agent=\"user\" and a task description.\n"
                          task branch-name))
          (gptel-send))
        (format "Task delegated to agent buffer '%s' in jj workspace.\nOrg entry updated to DELEGATED.\nYou can observe progress by visiting the agent buffer."
                child-buf-name)
        branch-name)))

** fanshi/clock-in-when-at-link-in-gtd-actions
#+begin_src emacs-lisp :tangle yes
(after! org
  (defun fanshi/clock-in-when-at-link-in-gtd-actions ()
    (save-excursion
      (when (derived-mode-p 'org-agenda-mode) (org-agenda-switch-to))
      (when (and (derived-mode-p 'org-mode)
                 (eq buffer-file-name (org-gtd--path org-gtd-default-file-name))
                 (eq 'link (org-element-type (org-element-context))))
        (org-back-to-heading)
        (let ((props (org-entry-properties)))
          (when (and (string= (alist-get "ORG_GTD" props nil nil #'equal) "Actions")
                     (string= (alist-get "TODO" props nil nil #'equal) "NEXT"))
            (org-clock-in)))))))
