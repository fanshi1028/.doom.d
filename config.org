(defun fanshi/clone-from-project (dir)
  "Clone the infrastructure of an existing project DIR to make a new project"
  (require 'projectile)
  (if-let ((project-type (projectile-project-type dir))
           (project-files-to-copy (symbol-value (intern-soft (concat "fanshi/project-files-to-copy/" (prin1-to-string project-type)))))
           (new-project-directory (file-name-as-directory (read-directory-name "Create new project at directory: "))))
      (progn
        ;; NOTE: make new directory
        (make-directory new-project-directory t)
        ;; NOTE: copy files
        (dolist (wildcards-or-file project-files-to-copy)
          (dolist (file (projectile-verify-file-wildcard wildcards-or-file dir))
                  (dired-copy-file file new-project-directory 1)))
        ;; NOTE: init project
        (fanshi/init-new-project new-project-directory)
        ;; NOTE: switch to the new project
        (projectile-switch-project-by-name new-project-directory))
    (user-error (concat "Unsupported project type to clones: " (prin1-to-string project) " + " (prin1-to-string project-type) " + " (prin1-to-string project-files-to-copy)))))

(after! gptel
  (defun fanshi/gptel--parse-llama-ini ()
    "Parse ~/.doom.d/llama-model.ini and return model configurations with multimodal support."
    (let ((models nil)
          (current-section nil)
          (current-props nil))
      (with-temp-buffer
        (insert-file-contents-literally
         (expand-file-name "llama-model.ini" doom-user-dir))
        (goto-char (point-min))
        (while (not (eobp))
          (let ((line (buffer-substring-no-properties
                       (line-beginning-position)
                       (line-end-position))))
            (cond
             ;; Section header: [model-name]
             ((string-match "^\\[\\([^]]+\\)\\]$" line)
              (when current-section
                (push (cons current-section current-props) models))
              (setq current-section (intern (match-string 1 line))
                    current-props nil))
             ;; Property line: key = value
             ((string-match "^\\([a-z-]+\\)\\s-*=\\s-*\\(.*\\)$" line)
              (let ((key (intern (concat ":" (match-string 1 line))))
                    (val (match-string 2 line)))
                (setq val (cond
                           ((string-match-p "^[0-9.]+$" val) (string-to-number val))
                           ((string= "nil" val) nil)
                           (t val)))
                ;; Build proper plist instead of alist
                (setq current-props (nconc current-props (list key val)))))
             ;; Ignore comments and empty lines
             ((or (string-match "^\\s-*;" line)
                  (string-match "^\\s-*$" line))
              nil))
            (forward-line 1)))
        ;; Save last section
        (when current-section
          (push (cons current-section current-props) models)))
      (nreverse models))))

#+begin_src emacs-lisp :tangle yes
(after! gptel
  (defun fanshi/gptel--infer-capabilities-from-hf (hf-tag)
    "Infer capabilities based on MODEL-NAME and HF tag."
    (let ((caps '())
          (mime-types nil)
          (model-name (downcase hf-tag)))
      ;; Check for Qwen3.5, GLM-4.7, gpt-oss patterns (multimodal)
      (when (or (string-match-p "qwen3\\.5" model-name)
                (string-match-p "glm-4\\.6" model-name)
                (string-match-p "glm-4\\.7" model-name)
                (string-match-p "gpt-oss" model-name))
        (setq caps '(media tool-use json url)
              mime-types '("image/jpeg" "image/png" "image/gif" "image/webp")))
      ;; Check for Qwen3-Coder, coding-specific models
      (when (or (string-match-p "-coder" model-name)
                (string-match-p "omnicoder" model-name))
        (setq caps '(tool-use json)))
      (cons caps mime-types))))

(after! gptel
  (defun fanshi/gptel--model-to-gptel-spec (model-name)
    "Convert MODEL-NAME to gptel model specification with capabilities."
    (when-let ((props (cdr (assoc model-name (fanshi/gptel--parse-llama-ini)))))
      ;; Infer capabilities from HF tag
      (let* ((hf-tag (or (plist-get props :hf) ""))
             (infer (fanshi/gptel--infer-capabilities-from-hf hf-tag))
             (caps (car infer))
             (mime-types (cdr infer)))
        `(,model-name
          :description ,(or (plist-get props :description) "Local Llama model")
          :capabilities ,caps
          :mime-types ,mime-types
          :context-window ,(or (plist-get props :ctx-size) 65536)
          :input-cost 0
          :output-cost 0
          :request-params ,(append
                            (when (plist-get props :temp) `(:temperature ,(plist-get props :temp)))
                            (when (plist-get props :top-p) `(:top_p ,(plist-get props :top-p)))
                            (when (plist-get props :top-k) `(:top_k ,(plist-get props :top-k)))
                            (when (plist-get props :min-p) `(:min_p ,(plist-get props :min-p)))
                            (when (plist-get props :presence-penalty)
                              `(:presence_penalty ,(plist-get props :presence-penalty)))))))))
